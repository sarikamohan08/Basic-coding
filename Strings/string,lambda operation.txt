str1=input()
list1=list(map(str,str1.split()))
print (list1)
str2=input()
list2=list(map(str,str2.split()))
print (list2)
power= list(map(lambda m,n: m+n,list1,list2))
print(power)
sum= list(map(lambda m,n: m+n,list1,list2))
print(sum)
sub=list(map(lambda m,n: m[0]+n[-2:]+m[1:],list1,list2))
print(sub)
mul=list(map(lambda m,n: m[0:1]+n[-1],list1,list2))
print(mul)
